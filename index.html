Base Funnel Query (Overall)

-- authId-level: collapse multiple loan applications per user via MAX()
WITH selected_offers AS (
  SELECT loanapplication_id, term AS selected_term
  FROM ued_qbf_dwh.match WHERE selected = true
),
bank_status AS (
  SELECT loanapplication_id,
    MAX(CASE WHEN status='COMPLETED' THEN 1 ELSE 0 END) as bank_completed
  FROM ued_qbf_dwh.applicantactionrequired
  WHERE action='VERIFY' AND subaction='BANK'
  GROUP BY loanapplication_id
),
BASE_LA AS (
  SELECT la.id AS la_id, u.authId,
    CASE WHEN (res.latestfraudandcompliancedecision='PRIMARY_AUTO_APPROVED'
               AND res.latestcreditdecision='PRIMARY_AUTO_APPROVED')
            OR aar_v.loanapplication_id IS NOT NULL
         THEN 1 ELSE 0 END AS is_contingent_approved,
    CASE WHEN dla.status='FINAL_APPROVED' OR la.status IN ('FUNDED','COMPLETED')
         THEN 1 ELSE 0 END AS is_final_approved,
    -- Implicit bank done: COMPLETED OR final/funded (retention policy fix)
    CASE WHEN bs.bank_completed=1 THEN 1
         WHEN dla.status='FINAL_APPROVED' OR la.status IN ('FUNDED','COMPLETED') THEN 1
         ELSE 0 END AS bank_done,
    so.selected_term,
    CASE WHEN l.fundeddate IS NOT NULL THEN 1 ELSE 0 END AS is_funded
  FROM ued_qbf_dwh.loanapplication la
  JOIN ued_qbf_dwh.directlendingloanapplication dla ON la.id=dla.LOANAPPLICATION_ID
  LEFT JOIN ued_qbf_dwh.loanapplicationuser lau ON la.id=lau.loanApplication_id
  LEFT JOIN ued_qbf_dwh.user u ON lau.user_id=u.id
  LEFT JOIN (SELECT * FROM (SELECT *, ROW_NUMBER() OVER (
    PARTITION BY directlendingloanapplication_id ORDER BY id DESC) AS rn
    FROM ued_qbf_dwh.ruleevaluationsummary) WHERE rn=1
  ) res ON dla.id=res.directlendingloanapplication_id
  LEFT JOIN ued_qbf_dwh.applicantactionrequired aar_v
    ON la.id=aar_v.loanapplication_id AND aar_v.action='VERIFY' AND aar_v.subaction='BANK'
  LEFT JOIN ued_qbf_dwh.loan l ON la.id=l.loanapplication_id
  LEFT JOIN selected_offers so ON la.id=so.loanapplication_id
  LEFT JOIN bank_status bs ON la.id=bs.loanapplication_id
  WHERE EXPERIENCE_ID=40 AND la.CREATEDDATE>='2025-09-18'
),
-- Collapse to one row per user
BASE AS (
  SELECT authId,
    MAX(is_contingent_approved) as is_contingent,
    MAX(is_final_approved)      as is_final,
    MAX(bank_done)              as bank_done,
    MAX(CASE WHEN selected_term IS NOT NULL THEN 1 ELSE 0 END) as has_offer,
    MAX(is_funded)              as is_funded,
    MIN(selected_term)          as selected_term
  FROM BASE_LA WHERE is_contingent_approved=1
  GROUP BY authId
),
uf AS (
  SELECT auth_id,
    MAX(fnpl_options_screen) os,
    MAX(CASE WHEN loan_option_selected_3_months=1 OR loan_option_selected_6_months=1
              OR loan_option_selected_9_months=1 THEN 1 ELSE 0 END) tc,
    MAX(add_bank_account) ab, MAX(fnpl_payment_schedule) ps,
    MAX(loan_accept) la, MAX(efile_attempt) ef
  FROM cgan_ustax_published.fnpl_funnel_first_touch_fact GROUP BY auth_id
)
SELECT COALESCE(CAST(b.selected_term AS STRING),'none') as term,
  COUNT(*)                                    as s1_contingent_BE,
  SUM(CASE WHEN f.os=1 THEN 1 ELSE 0 END)    as s2_options_CS,
  SUM(CASE WHEN f.tc=1 THEN 1 ELSE 0 END)    as s3_term_click_CS,
  SUM(CASE WHEN f.ab=1 THEN 1 ELSE 0 END)    as s4_add_bank_CS,
  SUM(b.bank_done)                            as s5_bank_done_BE,
  SUM(b.has_offer)                            as s6_offer_sel_BE,
  SUM(b.is_final)                             as s7_final_appr_BE,
  SUM(CASE WHEN f.ps=1 THEN 1 ELSE 0 END)    as s8_pmt_sched_CS,
  SUM(CASE WHEN f.la=1 THEN 1 ELSE 0 END)    as s9_loan_accept_CS,
  SUM(CASE WHEN f.la=1 AND f.ef=1 THEN 1 ELSE 0 END) as s10_accept_efile_CS,
  SUM(b.is_funded)                            as s11_funded_BE
FROM BASE b
LEFT JOIN uf f ON CAST(b.authId AS STRING)=f.auth_id
GROUP BY 1 ORDER BY 1
